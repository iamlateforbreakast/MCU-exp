#include <Arduino.h>
#include <U8g2lib.h>
#include <SPI.h>

// Constructor for SH1106 128x64 using Hardware SPI
// Format: U8G2_SH1106_128X64_NONAME_F_4W_HW_SPI(rotation, cs, dc [, reset])
U8G2_SH1106_128X64_NONAME_F_4W_HW_SPI u8g2(U8G2_R0, /* cs=*/ PA25, /* dc=*/ PA13, /* reset=*/ PA12);

void setup() {
  u8g2.begin();
}

void loop() {
  u8g2.clearBuffer();
  u8g2.setFont(u8g2_font_ncenB08_tr);
  u8g2.drawStr(0,10,"RTL8720DN SPI");
  u8g2.sendBuffer();
  delay(1000);
}
